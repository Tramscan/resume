% Custom commands
\newcommand{\sectionheading}[1]{%
  \vspace{0.3em}%
  {\large\textbf{\MakeUppercase{#1}}}
  \\[-0.2em]%
  \rule{\textwidth}{0.1pt}%
  \\[-0.1em]%
}

\newcommand{\entry}[2]{%
  \noindent\makebox[\textwidth]{%
    \parbox{\textwidth}{%
      \textbf{#1}\hfill\textit{#2}%
    }%
  }%
}
