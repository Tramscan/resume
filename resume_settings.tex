% Custom commands
\newcommand{\sectionheading}[1]{%
  \par\vspace{0.5em}% Add vertical space above the heading
  \noindent{\large\textbf{\MakeUppercase{#1}}}%
  \\[-0.85em]% Space between title and line
  \rule{\textwidth}{0.1pt}%
  \\[0.1em]% Space between line and text
}

\newcommand{\entry}[2]{%
  \noindent\makebox[\textwidth]{%
    \parbox{\textwidth}{%
      \textbf{#1}\hfill\textit{#2}%
    }%
  }%
}

\newcommand{\entryNoDate}[1]{%
  \noindent\makebox[\textwidth]{%
    \parbox{\textwidth}{%
      \textbf{#1}\hfill%
    }%
  }%
}
